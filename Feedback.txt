        for start_z in range(8, 0, -1):
            # Generate grid positions for this level specifically
            grid_positions = [(ox, oy) for ox in range(BASE_WIDTH) for oy in range(BASE_DEPTH)]
            self.rng.shuffle(grid_positions) # Randomize layout entry, keep height order

            for ox, oy in grid_positions:
                if self.timed_out():
                    break
                initial_sockets = [(ox, oy, start_z)]
                self.build(initial_sockets)
                
            # If we found a great high-altitude track, we can return it immediately
